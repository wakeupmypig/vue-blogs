# vuex原理

vuex是vue的状态管理工具,为了更方便实现多个组件共享状态

通过流程图来掌握vuex工作原理

import Vue from 'vue'
import Vuex from './vuex'
Vue.use(Vuex) // 默认会调用install方法
export default new Vuex.Store({
  state: { // 状态
    count:100
  },
  getters:{ // 计算属性
    getCount(state){
      return state.count+100;
    }
  },
  actions:{ // 异步更新
    change({commit}){
      setTimeout(() => {
        commit('change');
      }, 1000);
    }
  },
  mutations:{ // 同步更新
    change(state){
      state.count+=10;
    }
  }
})

class Store {
}
let Vue;
const install = _Vue => {
    Vue = _Vue;
    Vue.mixin({
        beforeCreate(){
           console.log(this.$options.name); // 会将每个组件的名字都打印出来
        }
    })
};
export default {
    Store,
    install

# $store.getters

class Store {
  constructor(options) {
    this.getters = {};
    this._vm = new Vue({
      data: {
        state: options.state
      }
    });
    let getters = options.getters;
    if(getters){
        Object.keys(getters).forEach(key=>{
            Object.defineProperty(this.getters,key,{ // 定义getters
                get:()=>{
                    return getters[key](this.state);
                }
            })
        })
    }
  }
  get state() {
    return this._vm.state;
  }
}

# commit & dispatch方法的处理

因为在action中会解构commit和dispatch来使用，这时候this执行会发生变化

let {commit,dispath} = this;
// 在实例上增加commit 和 dispatch 方法
this.commit = (type) =>{
	commit.call(this,type);
}
this.dispath = (type) =>{
	dispath.call(this,type);
}

# 收集模块

this.modules = new ModuleCollection(options);
class ModuleCollection {
  constructor(options) {
    this.register([], options);
  }
  register(path, rootModule) {
    let newModule = {  // 格式化 传入的对象，转化成树形结构
      _raw: rootModule,
      _children: {},
      state: rootModule.state
    };
    if(path.length === 0){ // 如果path.length == 0 那么就是根模块
        this.root = newModule;
    }else{
       // 子模块
        let parent = path.slice(0,-1).reduce((memo,current)=>{
            return memo._children[current];
        },this.root);
        parent._children[path[path.length-1]] = newModule;
    }
    if(rootModule.modules){ // 是子模块就递归注册
        forEach(rootModule.modules,(moduleName,value)=>{
            this.register(path.concat(moduleName),value);
        });
    }
  }
}

# 装载模块

const installModule = (store,state,path,rootModule)=>{
    if(rootModule._raw.getters){ // 将getter定义到store上
        forEach(rootModule._raw.getters,(getterName,getterFn)=>{
            Object.defineProperty(store.getters,getterName,{
                get:()=>{
                    return getterFn(rootModule.state);
                }
            })
        })
    }
    if(rootModule._raw.mutations){
        forEach(rootModule._raw.mutations,(mutationName,mutationFn)=>{
            store.mutations[mutationName] = (store.mutations[mutationName] = []);
            store.mutations[mutationName].push(()=>{
                mutationFn.call(store,state);
            });
        })
    }
    if(rootModule._raw.actions){
        forEach(rootModule._raw.actions,(actionName,actionFn)=>{
            store.actions[actionName] = (store.actions[actionName] = []);
            store.actions[actionName].push(function(){
                actionFn.call(store,this);
            });
        })
    }
}
installModule(this,this.state,[],this.modules.root);


dispatch(type) {
    this.actions[type].forEach(fn=>fn.call(this));
 }
 commit(type) {
    this.mutations[type].forEach(fn=>fn.call(this));
 }

const mapState = arrList => {
  let obj = {};
  for (let i = 0; i < arrList.length; i++) {
    let stateName = arrList[i];
    obj[stateName] = function() {
      return this.$store.state[stateName];
    };
  }
  return obj;
};
