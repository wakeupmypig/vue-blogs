# 作业及npm的使用

# 1) 柯里化函数

函数分步传递参数，将函数拆分成功能更具体化的函数

const add = (a, b, c, d) => {
  return a + b + c + d;
};
const currying = (fn, args = []) => {
  // args是每次调用时收集的参数列表
  let len = fn.length; //函数执行的参数列表
  return (..._) => { 
    args.push(..._); // 将参数手机好
    if (args.length < len) {
      return currying(fn, args);
    }
    return fn(...args); // 当参数个数达到时执行此函数
  };
};
let r = currying(add)(1)(2, 3)(4);

反柯里化的作用则在于扩大函数的适用性

const uncurrying = fn => obj =>{
    return fn.call(...obj);
}
const check = uncurrying(Object.prototype.toString);

Function.prototype.uncurrying = function(){
    return (...args)=>{
        return this.call(...args); // Function.prototype.call.apply(self,arguments);
    }
}
let type = Object.prototype.toString.uncurrying();
let r = type('hello');

# 4) Mixin的实现

let mixin = (superClass) =>{
    return class Proxy extends superClass{
        beforeCreate(){
            console.log('middle')
            super.beforeCreate();
        }
    }
}
class Parent{
    beforeCreate(){
        console.log('parent');
    }
}
class Child extends mixin(Parent){
    beforeCreate(){
        console.log('child');
        super.beforeCreate()
    }
}
let child = new Child();
child.beforeCreate();

# 5) 数组的flatten

Array.prototype.flat = function(n = 1){
    if(n == 0){return this}
    return this.reduce((a,b)=>{
       if(Array.isArray(b)){
           return a.concat(b.flat(--n));
       }else{
           return [...a,b];
       }
    },[]);
}
console.log([1,[2,[3,[4,[5]]]]].flat(3));

# 必备知识点

# 1) node中的模块

  • 模块分类 ES6module,commonjs规范 amd,cmd,umd
  • commonjs规范
    • 一个文件就是一个模块
    • 如果模块想给别人使用 module.exports / exports 同一个对象但是最终导出的是module.exports
    • 如果想使用这个模块 require (同步读取文件，包一个自执行函数，vm.runInthisContext,传入export对象，最终返回的是exports 对象，所以就可以拿到其他模块的内容)
  • 模块的查找规范
    • 第三方模块 module.paths
  • 如果文件和文件夹重名 先取文件，文件不能取到，找文件夹 package.json => main => index.js
