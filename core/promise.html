# 柯里化函数

const currying = (fn,args = []) => {
    let len = fn.length;
    return (..._)=>{
        args = args.concat(_);
        if(args.length < len){
            return currying(fn,args);
        }
        return fn(...args);
    }
};
const add = (a, b, c, d, e) => {
  return a + b + c + d + e;
};
let r = currying(add)(1)(2,3)(4,5);
console.log(r);
