# 类的基本使用

声明实例属性以及公共方法

class Animal {
    public type: string; // 实例上的属性
    constructor(type: string) {
        this.type = type;
    }
    public getType() { // 公共方法
        return this.type;
    }
}
let animal = new Animal('哺乳类');
console.log(animal.type);

class Animal {
    public type: string;
    constructor(type: string) {
        this.type = type;

    public getType() {
        return this.type;
    }
}
class Tiger extends Animal {
    public name: string;
    constructor(type: string, name: string) {

let tiger = new Tiger('哺乳类', '大王');
console.log(tiger);

# 类的修饰符 (4钟)

  • 修饰符用来修饰属性和方法
  • constructor如果被private修饰那么这个类不能被扩展
  • constructor如果被protected修饰那么这个类不能被实例化
  • eadonly不能修饰constructor

# public

公开的:用public来修饰的可以在当前类和继承自己的类以及在类外面调用

class Animal {
    public type: string;
    constructor(type: string) {
        this.type = type;
    }
    public getType() {
        return this.type;
    }
}
class Tiger extends Animal {
    public name: string;
    constructor(type: string, name: string) {
        super(type);
        console.log(this.type); // 在子类中可以访问
        this.name = name;
    }
}
let tiger = new Tiger('哺乳类', '大王');
console.log(tiger.type); // 在外面也可以被访问

# private

        return this.type;

}
class Tiger extends Animal {
    public name: string;

# readonly

仅读属性表示外部不可以更改此属性

class Animal {
    public readonly type: string;
    constructor(type: string) {
        this.type = type;
    }
    public getType() {
        return this.type;
    }
}
let animal = new Animal("软体动物");
console.log(animal.type);
animal.type = '哺乳动物' // Cannot assign to 'type' because it is a read-only

# 参数属性

可以在构造函数中的参数增加任意修饰符，将属性转化成实例上的属性

class Animal {
    constructor(readonly type: string) { }
    public getType() {
        return this.type;

}
let animal = new Animal("软体动物");

# 可选属性

实例化类时，可以标识某个属性不用必须传递，和函数的可选参数类似

class Animal {
    // 如果age没有传递属性值默认为undefined
    constructor(public type: string, public age?: string) {
    }
}
let animal = new Animal('哺乳类');
console.log(animal); // Animal {type: "哺乳类", age: undefined}

# 属性访问器

通过属性访问器对类中的某个属性进行操作

class Animal {
    private _myName: string;
    public get name(): string {
        return this._myName;
    }
    public set name(value) { // set方法不能对返回类型做批注
        this._myName = value;
    }
}
let animal = new Animal();
animal.name = '动物'
console.log(animal.name);

# 抽象类

通过abstract属性描述类或者类中的属性,通过abstract来描述的类不能被实例化

abstract class Animal {
    public abstract type: string;
    public abstract getType(): void; // 兼容性的问题？？？
}
class Tiger extends Animal {
    public type: string;
    public getType(): string {
        return this.type
    }
}
let tiger = new Tiger();
console.log(tiger.getType())
