# 接口

接口就是用来给代码定义结构，定义代码的规则，接口中的方法都是抽象的，不能有具体的实现

# 基本使用

可以用来限制对象类型

interface ISchool {
    name: string,
    age: number
}
// 限制对象
let school: ISchool = {
    name: 'zf',
    age: 10
}
// 限制函数的参数
let getSchool = ({ name, age }: ISchool): string => {
    return `${name} ${age}`
}
getSchool(school);

# 可选参数

通过? 来描述此属性非必传属性

interface ISchool {
    name: string,
    age?: number
}
let getSchool = ({ name, age = 10 }: ISchool): string => {
    return `${name} ${age}`
}
getSchool({ name: 'zf' });

# 接口属性限制

限制接口里必须有哪些属性，其他属性任意即可，也可以使用类型断言强制断言要传递的属性

interface ISchool {
    name: string,
    age?: number,
    [propName: string]: any
}

let school: ISchool = {
    name: 'zf',
    age: 10,
    address: '回龙观东大街'
}

    age: 10,

# 限定类

类的继承只能有一个父类但是可以实现多个接口,类表示一种类型，接口表示一类行为

interface IFly {
    fly(): void
}
interface IRun {
    run(): void
}
class My implements IFly, IRun {
    fly() { }
    run() { }
}

# 接口继承

同一个接口可以继承多个接口

interface IAnimal {
    speak(): void
}
interface ITiger extends IAnimal {
    eat(): void
}
class Tiger implements ITiger {
    speak() { }

# 约束类的构造函数

class Animal {
    constructor(public name: string) { }
}
const create = (a: new (name: string) => Animal, name: string): Animal => {
    return new a(name);
}
create(Animal, '动物');

// 使用接口定义
interface ICtor {
    new(name: string): Animal
}
class Animal {
    constructor(public name: string) { }
}
const create = (a: ICtor, name: string): Animal => {
    return new a(name);
}
console.log(create(Animal, '动物'));
