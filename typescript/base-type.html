# 枚举类型 enum

枚举类型让我们的代码更具有可读性

enum Rules {
    ADMIN,
    TEACHER,
    STUDENT
}
console.log(Rules.ADMIN); // 默认序号从0开始

enum Rules {
    ADMIN = 100,
    TEACHER,
    STUDENT
}
console.log(Rules.TEACHER); // 手动指定开始值

enum Rules {
    ADMIN = 1,
    TEACHER = 5,
    STUDENT
}
console.log(Rules[5]); // 反向取值

// 原理
let Rules: any = {};
Rules[Rules['ADMIN'] = 1] = 'ADMIN'
Rules[Rules['TEACHER'] = 5] = 'TEACHER'
Rules[Rules['STUDENT'] = 6] = 'STUDENT'
console.log(Rules);

# 类型断言

function getLen(val: string | number) {
    if ((val as string).length) {
        return (<string>val).length
    } else {
        return val.toString().length
    }
}
