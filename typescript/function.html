# 函数

声明函数的两种方式:函数声明、表达式声明

# 函数声明

function say(message: string): string {
    return message;
}

# 类型别名

使用type关键字，定义自己的类型,可以用来限定类型或者值,可以复用类型

type one = 1 | 'first' | 'one';
let val: one = 1;

// 可以给函数类型起别名

type say = (m: string) => string;
let say: say = (message: string): string => {
    return message

# 函数的可选参数

可选参数必须位于必选参数之后 (使用?)

function say(name: string, age?: number): string {
    if (typeof age == 'undefined') {
        return name
    } else {
        return name + age;
    }
}

# 函数的默认参数

默认参数可以放到任意位置，传入的值为undefined时使用默认参数

function say(name: string = 'hello', age: number): string {
    return name + age;
}
say(undefined, 18)

# 函数的剩余参数

剩余运算符只能用到参数的末尾

function sum(currency: string, ...args: number[]): string {
    return currency + args.reduce((prev: number, current: number): number => {
        return prev + current
    })
}
sum('$', 1, 2, 3, 4, 5);

# 函数的重载

function toArray(value: string): string[];
function toArray(value: number): number[];
function toArray(value: any): any {
    if (typeof value === 'string') {
        return value.split('');
    } else {
        return Array.of(value);
    }
}
console.log(toArray(3)); // 限制返回值的类型
